{{- /* 图片渲染钩子：WebP 转换 + srcset + lazy loading */ -}} {{- $src := .Destination | safeURL -}} {{- $alt := .Text | plainify -}} {{- $title := .Title -}} {{- $isExternal := or (strings.HasPrefix $src "http://") (strings.HasPrefix $src "https://") -}} {{- if $isExternal -}} {{- else -}} {{- $img := .Page.Resources.GetMatch $src -}} {{- if $img -}} {{- $webp := $img.Resize (printf "%dx webp q85" $img.Width) -}} {{- $w480 := $img.Resize "480x webp q80" -}} {{- $w720 := $img.Resize "720x webp q80" -}} {{- $w1080 := $img.Resize "1080x webp q80" -}} {{- else -}} {{- end -}} {{- end -}}
